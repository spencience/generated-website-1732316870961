document.addEventListener('DOMContentLoaded', function() {
    const syntaxInput = document.getElementById('syntax');
    const codeInput = document.getElementById('code');
    const preview = document.querySelector('#preview pre');
    const form = document.querySelector('form');

    // Auto-resize textareas
    function autoResize(element) {
        element.style.height = 'auto';
        element.style.height = element.scrollHeight + 'px';
    }

    [syntaxInput, codeInput].forEach(textarea => {
        textarea.addEventListener('input', () => {
            autoResize(textarea);
            updatePreview();
        });
    });

    // Live preview functionality
    function updatePreview() {
        const syntax = syntaxInput.value.trim();
        const code = codeInput.value.trim();
        
        if (syntax && code) {
            preview.innerHTML = `${syntax}\n\n${code}`
                .replace(/@syntaxexion\b/g, '<span class="text-yellow-400">@syntaxexion</span>')
                .replace(/pattern:/g, '<span class="text-pink-400">pattern:</span>')
                .replace(/expansion:/g, '<span class="text-green-400">expansion:</span>')
                .replace(/(\$[a-zA-Z]+)/g, '<span class="text-blue-400">$1</span>');
        } else {
            preview.textContent = 'Enter both syntax pattern and test code to see preview';
        }
    }

    // Form validation
    form.addEventListener('submit', function(e) {
        if (!syntaxInput.value.trim() || !codeInput.value.trim()) {
            e.preventDefault();
            alert('Please fill in both syntax pattern and test code fields');
        }
    });

    // Copy functionality
    preview.addEventListener('click', async function() {
        try {
            await navigator.clipboard.writeText(this.textContent);
            this.classList.add('bg-indigo-700');
            setTimeout(() => this.classList.remove('bg-indigo-700'), 200);
        } catch (err) {
            console.error('Failed to copy:', err);
        }
    });

    // Mobile navigation
